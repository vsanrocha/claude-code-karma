# File Watcher Service v0

Design document for a Go-based file watcher service that monitors `~/.claude/` for real-time session updates.

## Goals

1. **Real-time monitoring** — Detect new sessions, message appends, and subagent spawns as they happen
2. **Parallel parsing** — Parse multiple JSONL files concurrently using goroutines
3. **Learn Go concurrency** — Hands-on with channels, worker pools, context cancellation
4. **Coexist with Python** — Python handles historical data, Go handles live updates

## Library Decision: In-House vs fswatcher

We evaluated **[github.com/sgtdi/fswatcher](https://github.com/sgtdi/fswatcher)**, a production-ready file watching library, but decided to **build in-house** using `fsnotify` directly.

### Why Not fswatcher?

| Concern | Impact |
|---------|--------|
| **Debouncing delays** | 100ms cooldown conflicts with "detect as they happen" goal |
| **Learning goals** | Document explicitly states "Learn Go concurrency" — fswatcher abstracts this away |
| **Custom classification** | Need domain-specific `FileKind` enum (Session/Subagent/ToolResult) |
| **Delta parsing** | `SessionTracker` with file offsets is unique to JSONL streaming |
| **Directory-centric API** | Need file-level granularity for precise tracking |

### What We're Building In-House

1. **Immediate events** — No debouncing, sub-millisecond latency from write to WebSocket broadcast
2. **Custom event router** — `classify()` function with `FileKind` enum for precise routing
3. **Session delta parser** — Tail-like behavior with offset tracking for JSONL files
4. **Worker pool pattern** — Hands-on with channels, goroutines, and fan-in/fan-out
5. **Domain-specific** — Built for `~/.claude/` structure, not generic file watching

This approach trades ~200 lines of extra code for:
- Complete control over event timing
- Educational value (hands-on Go concurrency)
- Domain-specific optimizations (JSONL delta parsing)

### Core Components

#### 1. File Watcher (fsnotify + recursive)

```go
// internal/watcher/watcher.go
package watcher

type ClaudeWatcher struct {
    watcher    *fsnotify.Watcher
    events     chan FileEvent
    errors     chan error
    root       string
    ctx        context.Context
    cancel     context.CancelFunc
}

type FileEvent struct {
    Type      EventType  // Created, Modified, Deleted
    Path      string
    SessionID string     // extracted from path
    FileKind  FileKind   // Session, Subagent, ToolResult, Todo
}

type FileKind int

const (
    KindSession FileKind = iota
    KindSubagent
    KindToolResult
    KindTodo
    KindDebugLog
)

func NewClaudeWatcher(root string) (*ClaudeWatcher, error) {
    fsw, err := fsnotify.NewWatcher()
    if err != nil {
        return nil, err
    }

    ctx, cancel := context.WithCancel(context.Background())

    cw := &ClaudeWatcher{
        watcher: fsw,
        events:  make(chan FileEvent, 100),
        errors:  make(chan error, 10),
        root:    root,
        ctx:     ctx,
        cancel:  cancel,
    }

    // Add all existing directories recursively
    if err := cw.addRecursive(root); err != nil {
        return nil, err
    }

    go cw.run()
    return cw, nil
}

func (cw *ClaudeWatcher) run() {
    for {
        select {
        case <-cw.ctx.Done():
            return
        case event := <-cw.watcher.Events:
            if fe, ok := cw.classify(event); ok {
                cw.events <- fe
            }
            // Watch new directories
            if event.Op&fsnotify.Create != 0 {
                if info, err := os.Stat(event.Name); err == nil && info.IsDir() {
                    cw.watcher.Add(event.Name)
                }
            }
        case err := <-cw.watcher.Errors:
            cw.errors <- err
        }
    }
}

func (cw *ClaudeWatcher) classify(event fsnotify.Event) (FileEvent, bool) {
    path := event.Name

    fe := FileEvent{Path: path}

    switch {
    case event.Op&fsnotify.Create != 0:
        fe.Type = EventCreated
    case event.Op&fsnotify.Write != 0:
        fe.Type = EventModified
    case event.Op&fsnotify.Remove != 0:
        fe.Type = EventDeleted
    default:
        return fe, false
    }

    // Classify file kind based on path patterns
    switch {
    case strings.HasSuffix(path, ".jsonl") && strings.Contains(path, "/subagents/"):
        fe.FileKind = KindSubagent
        fe.SessionID = extractSessionID(path)
    case strings.HasSuffix(path, ".jsonl"):
        fe.FileKind = KindSession
        fe.SessionID = extractSessionIDFromFile(path)
    case strings.Contains(path, "/tool-results/"):
        fe.FileKind = KindToolResult
        fe.SessionID = extractSessionID(path)
    case strings.Contains(path, "/todos/"):
        fe.FileKind = KindTodo
    case strings.Contains(path, "/debug/"):
        fe.FileKind = KindDebugLog
    default:
        return fe, false
    }

    return fe, true
}
```

#### 2. Worker Pool (fan-out/fan-in)

```go
// internal/worker/pool.go
package worker

type Pool struct {
    workers   int
    jobs      chan Job
    results   chan Result
    wg        sync.WaitGroup
}

type Job struct {
    Event     FileEvent
    Priority  int
}

type Result struct {
    SessionID string
    Data      interface{}  // Parsed session, message delta, etc.
    Error     error
}

func NewPool(workers int) *Pool {
    p := &Pool{
        workers: workers,
        jobs:    make(chan Job, 100),
        results: make(chan Result, 100),
    }

    for i := 0; i < workers; i++ {
        p.wg.Add(1)
        go p.worker(i)
    }

    return p
}

func (p *Pool) worker(id int) {
    defer p.wg.Done()

    for job := range p.jobs {
        result := p.process(job)
        p.results <- result
    }
}

func (p *Pool) process(job Job) Result {
    switch job.Event.FileKind {
    case KindSession:
        return p.parseSessionDelta(job.Event)
    case KindSubagent:
        return p.parseSubagent(job.Event)
    case KindToolResult:
        return p.parseToolResult(job.Event)
    default:
        return Result{Error: fmt.Errorf("unknown file kind")}
    }
}

func (p *Pool) Submit(job Job) {
    p.jobs <- job
}

func (p *Pool) Results() <-chan Result {
    return p.results
}

func (p *Pool) Shutdown() {
    close(p.jobs)
    p.wg.Wait()
    close(p.results)
}
```

#### 3. Session Delta Parser (tail-like)

```go
// internal/parser/session.go
package parser

type SessionTracker struct {
    mu       sync.RWMutex
    offsets  map[string]int64  // sessionID -> last read offset
}

func NewSessionTracker() *SessionTracker {
    return &SessionTracker{
        offsets: make(map[string]int64),
    }
}

// ParseDelta reads only new lines since last read
func (st *SessionTracker) ParseDelta(path string) ([]Message, error) {
    st.mu.Lock()
    offset := st.offsets[path]
    st.mu.Unlock()

    file, err := os.Open(path)
    if err != nil {
        return nil, err
    }
    defer file.Close()

    // Seek to last position
    if offset > 0 {
        file.Seek(offset, io.SeekStart)
    }

    var messages []Message
    scanner := bufio.NewScanner(file)

    for scanner.Scan() {
        line := scanner.Bytes()
        if msg, err := parseMessage(line); err == nil {
            messages = append(messages, msg)
        }
    }

    // Update offset
    newOffset, _ := file.Seek(0, io.SeekCurrent)
    st.mu.Lock()
    st.offsets[path] = newOffset
    st.mu.Unlock()

    return messages, nil
}

func parseMessage(data []byte) (Message, error) {
    var raw map[string]interface{}
    if err := json.Unmarshal(data, &raw); err != nil {
        return Message{}, err
    }

    msgType, _ := raw["type"].(string)

    switch msgType {
    case "user":
        var msg UserMessage
        json.Unmarshal(data, &msg)
        return Message{Type: "user", User: &msg}, nil
    case "assistant":
        var msg AssistantMessage
        json.Unmarshal(data, &msg)
        return Message{Type: "assistant", Assistant: &msg}, nil
    default:
        return Message{}, fmt.Errorf("unknown message type: %s", msgType)
    }
}
```

#### 4. WebSocket Server

```go
// internal/server/websocket.go
package server

type Hub struct {
    clients    map[*Client]bool
    broadcast  chan []byte
    register   chan *Client
    unregister chan *Client
    mu         sync.RWMutex
}

type Client struct {
    hub      *Hub
    conn     *websocket.Conn
    send     chan []byte
    filters  ClientFilters  // subscribe to specific sessions/projects
}

type ClientFilters struct {
    SessionIDs []string
    ProjectIDs []string
    AllEvents  bool
}

type WSMessage struct {
    Type      string      `json:"type"`       // "session_update", "new_message", "subagent_spawned"
    SessionID string      `json:"sessionId"`
    ProjectID string      `json:"projectId"`
    Data      interface{} `json:"data"`
    Timestamp time.Time   `json:"timestamp"`
}

func (h *Hub) Run(results <-chan worker.Result) {
    for {
        select {
        case client := <-h.register:
            h.mu.Lock()
            h.clients[client] = true
            h.mu.Unlock()

        case client := <-h.unregister:
            h.mu.Lock()
            if _, ok := h.clients[client]; ok {
                delete(h.clients, client)
                close(client.send)
            }
            h.mu.Unlock()

        case result := <-results:
            msg := WSMessage{
                Type:      resultToType(result),
                SessionID: result.SessionID,
                Data:      result.Data,
                Timestamp: time.Now(),
            }
            data, _ := json.Marshal(msg)

            h.mu.RLock()
            for client := range h.clients {
                if client.shouldReceive(msg) {
                    select {
                    case client.send <- data:
                    default:
                        // Client too slow, disconnect
                        close(client.send)
                        delete(h.clients, client)
                    }
                }
            }
            h.mu.RUnlock()
        }
    }
}
```

#### 5. Main Entry Point

```go
// cmd/watcher/main.go
package main

func main() {
    // Configuration
    claudeDir := filepath.Join(os.Getenv("HOME"), ".claude")
    port := flag.Int("port", 8081, "WebSocket server port")
    workers := flag.Int("workers", runtime.NumCPU(), "Number of parser workers")
    flag.Parse()

    // Setup context for graceful shutdown
    ctx, cancel := signal.NotifyContext(context.Background(), os.Interrupt, syscall.SIGTERM)
    defer cancel()

    // Initialize components
    watcher, err := watcher.NewClaudeWatcher(claudeDir)
    if err != nil {
        log.Fatal(err)
    }

    pool := worker.NewPool(*workers)
    hub := server.NewHub()

    // Wire up the pipeline
    go func() {
        for event := range watcher.Events() {
            pool.Submit(worker.Job{Event: event})
        }
    }()

    go hub.Run(pool.Results())

    // Start HTTP/WebSocket server
    http.HandleFunc("/ws", hub.HandleWebSocket)
    http.HandleFunc("/health", func(w http.ResponseWriter, r *http.Request) {
        json.NewEncoder(w).Encode(map[string]string{"status": "ok"})
    })

    server := &http.Server{Addr: fmt.Sprintf(":%d", *port)}

    go func() {
        log.Printf("File watcher listening on :%d", *port)
        if err := server.ListenAndServe(); err != http.ErrServerClosed {
            log.Fatal(err)
        }
    }()

    // Wait for shutdown
    <-ctx.Done()
    log.Println("Shutting down...")

    watcher.Close()
    pool.Shutdown()
    server.Shutdown(context.Background())
}
```

### Future WebSocket Hook (Phase 5)

**New Hook to Add:**

```typescript
// apps/web/hooks/use-session-stream.ts (FUTURE)
import { useEffect, useRef } from 'react';
import { useQueryClient } from '@tanstack/react-query';

interface WSMessage {
  type: 'session_update' | 'new_message' | 'subagent_spawned' | 'todo_update';
  sessionId: string;
  projectId: string;
  data: unknown;
  timestamp: string;
}

export function useSessionStream(sessionId?: string) {
  const wsRef = useRef<WebSocket | null>(null);
  const queryClient = useQueryClient();

  useEffect(() => {
    const ws = new WebSocket('ws://localhost:8081/ws');
    wsRef.current = ws;

    ws.onopen = () => {
      // Subscribe to specific session or all events
      ws.send(JSON.stringify({
        type: 'subscribe',
        sessionIds: sessionId ? [sessionId] : [],
        allEvents: !sessionId,
      }));
    };

    ws.onmessage = (event) => {
      const msg: WSMessage = JSON.parse(event.data);

      // Invalidate TanStack Query cache for affected data
      switch (msg.type) {
        case 'new_message':
          queryClient.invalidateQueries({
            queryKey: ['session', msg.sessionId]
          });
          break;
        case 'session_update':
          queryClient.invalidateQueries({
            queryKey: ['session', msg.sessionId]
          });
          queryClient.invalidateQueries({
            queryKey: ['projects']  // May affect project stats
          });
          break;
        case 'subagent_spawned':
          queryClient.invalidateQueries({
            queryKey: ['session', msg.sessionId, 'subagents']
          });
          break;
        case 'todo_update':
          queryClient.invalidateQueries({
            queryKey: ['session', msg.sessionId, 'todos']
          });
          break;
      }
    };

    ws.onerror = (error) => {
      console.error('WebSocket error:', error);
    };

    return () => {
      ws.close();
    };
  }, [sessionId, queryClient]);
}
```
